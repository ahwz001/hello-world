## 对于 Linux ##
sudo pip install sth
# 或者明确版本
sudo pip2 install sth
sudo pip3 install sth
sudo python2 -m pip install sth
sudo /path/to/python -m pip install sth

## 对于 Windows NT ##

# 如果仅安装 python2
pip install sth
# 如果安装有 python3, 则需要明确 pip 版本
py -2 -m pip install sth
py -3 -m pip install sth
